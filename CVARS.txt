script "GiveAllPlayersRedCard" (void) NET
{
    for (int p = 0; p < 8; p++)
    {
        if (PlayerInGame(p) && !CheckActorInventory(800 + p, "RedCard"))
        {
            GiveActorInventory(800 + p, "RedCard", 1);
        }
    }
}
